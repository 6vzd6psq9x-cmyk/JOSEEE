    <!-- Carga de Tailwind CSS -->
    <script src="https://cdn.tailwindcss.com"></script>
    
    <script>
        // Configuración de Tailwind para usar la fuente Inter
        tailwind.config = {
            theme: {
                extend: {
                    fontFamily: {
                        inter: ['Inter', 'sans-serif'],
                    },
                },
            }
        }
    </script>
    
    <!-- Carga de React y Babel para JSX -->
    <script src="https://unpkg.com/react@18/umd/react.development.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    
    <style>
        /* Asegura que el contenedor principal ocupe toda la pantalla y elimina márgenes */
        body, #root {
            margin: 0;
            padding: 0;
            width: 100vw;
            height: 100vh;
            overflow: hidden; 
        }

        /* Estilo para el efecto de sombra en el botón de la IA */
        .gemini-shadow {
            box-shadow: 0 0 15px rgba(255, 255, 255, 0.5), 0 0 5px rgba(255, 255, 255, 0.8);
        }
    </style>

<body>

    <div id="root"></div>

    <script type="text/babel">
        const { useState, useCallback } = React;

        // Configuration for Gemini API (empty API key is handled by the canvas environment)
        const GEMINI_MODEL = "gemini-2.5-flash-preview-09-2025";
        const API_KEY = ""; // Placeholder, will be replaced by the environment

        // Lista de clases de color de fondo de Tailwind para el cambio aleatorio
        const BACKGROUND_COLORS = [
            'bg-red-700', 'bg-blue-700', 'bg-green-700', 'bg-yellow-700',
            'bg-purple-700', 'bg-pink-700', 'bg-indigo-700', 'bg-teal-700',
            'bg-orange-700', 'bg-gray-800'
        ];

        // Icono SVG para Configuración (Rueda dentada)
        const SettingsIcon = (props) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24"
                fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round"
                strokeLinejoin="round" {...props}>
                <path d="M12.22 2h-.44a2 2 0 0 0-2 2v.18a2 2 0 0 1-1 1.73l-.43.25a2 2 0 0 1-2 0l-.15-.08a2 2 0 0 0-2.73.73l-.78 1.35a2 2 0 0 0 .72 2.73l.15.08a2 2 0 0 1 1 1.73v.44a2 2 0 0 1-1 1.73l-.15.08a2 2 0 0 0-.73 2.73l.78 1.35a2 2 0 0 0 2.73.73l.15-.08a2 2 0 0 1 2 0l.43.25a2 2 0 0 1 1 1.73v.18a2 2 0 0 0 2 2h.44a2 2 0 0 0 2-2v-.18a2 2 0 0 1 1-1.73l.43-.25a2 2 0 0 1 2 0l.15.08a2 2 0 0 0 2.73-.73l.78-1.35a2 2 0 0 0-.72-2.73l-.15-.08a2 2 0 0 1-1-1.73v-.44a2 2 0 0 1 1-1.73l.15-.08a2 2 0 0 0 .73-2.73l-.78-1.35a2 2 0 0 0-2.73-.73l-.15.08a2 2 0 0 1-2 0l-.43-.25a2 2 0 0 1-1-1.73V4a2 2 0 0 0-2-2z"/><circle cx="12" cy="12" r="3"/>
            </svg>
        );

        // Icono de Estrella/Recompensa
        const StarIcon = (props) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none"
                stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" {...props}>
                <polygon points="12 2 15.09 8.26 22 9.27 17 14.14 18.18 21.02 12 17.77 5.82 21.02 7 14.14 2 9.27 8.91 8.26 12 2"/>
            </svg>
        );

        // Componente principal de la aplicación de Contador
        const App = () => {
            const [count, setCount] = useState(0); 
            const [backgroundColor, setBackgroundColor] = useState('bg-slate-900'); 
            const [isSettingsOpen, setIsSettingsOpen] = useState(false); 
            const [isColorChangeEnabled, setIsColorChangeEnabled] = useState(true); 
            const [rewardMessage, setRewardMessage] = useState(''); // Mensaje de la IA
            const [isLoading, setIsLoading] = useState(false); // Estado de carga de la IA
            const [isModalOpen, setIsModalOpen] = useState(false); // Estado del modal de la IA

            // Función para obtener un nuevo color aleatorio que no sea el actual
            const getNewRandomColor = useCallback(() => {
                let newColor = backgroundColor;
                while (newColor === backgroundColor) {
                    const randomIndex = Math.floor(Math.random() * BACKGROUND_COLORS.length);
                    newColor = BACKGROUND_COLORS[randomIndex];
                }
                return newColor;
            }, [backgroundColor]);

            // Maneja el cambio manual de color de fondo
            const handleManualColorChange = useCallback(() => {
                setBackgroundColor(getNewRandomColor());
            }, [getNewRandomColor]);

            // Maneja el incremento del contador y el cambio de color automático
            const handleIncrement = useCallback(() => {
                setCount(prevCount => prevCount + 1);
                if (isColorChangeEnabled) {
                    setBackgroundColor(getNewRandomColor());
                }
            }, [isColorChangeEnabled, getNewRandomColor]);

            // Reinicia el contador a cero y cierra la configuración
            const handleReset = useCallback(() => {
                setCount(0);
                setIsSettingsOpen(false);
            }, []);

            // Alterna la función de cambio de color automático
            const handleToggleColorChange = useCallback(() => {
                setIsColorChangeEnabled(prev => !prev);
            }, []);

            // Función para llamar a la API de Gemini
            const getAIPrompt = async (currentCount) => {
                setIsLoading(true);
                setIsModalOpen(true);
                setRewardMessage("Generando meta de clics... ✨");

                // System Instruction: Guía al modelo para actuar como un motivador lúdico
                const systemPrompt = "Actúa como un entrenador de juego motivacional. Basado en el número de clics que se te da, genera un mensaje de felicitación y una meta lúdica y concisa para el próximo hito. El tono debe ser divertido y alentador.";
                
                // User Query: La tarea específica para el modelo
                const userQuery = `El contador actual es ${currentCount}. Crea un mensaje de felicitación por el progreso y sugiere un reto para el siguiente hito significativo.`;

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/${GEMINI_MODEL}:generateContent?key=${API_KEY}`;
                
                const payload = {
                    contents: [{ parts: [{ text: userQuery }] }],
                    systemInstruction: { parts: [{ text: systemPrompt }] },
                };

                let attempt = 0;
                const maxAttempts = 3;
                let responseText = "No se pudo conectar con la IA. Inténtalo de nuevo más tarde.";

                while (attempt < maxAttempts) {
                    try {
                        const response = await fetch(apiUrl, {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json' },
                            body: JSON.stringify(payload)
                        });

                        if (response.ok) {
                            const result = await response.json();
                            const candidate = result.candidates?.[0];
                            if (candidate && candidate.content?.parts?.[0]?.text) {
                                responseText = candidate.content.parts[0].text;
                            }
                            break; // Sale del bucle si es exitoso
                        } else {
                            throw new Error(`API returned status ${response.status}`);
                        }
                    } catch (error) {
                        console.error(`Attempt ${attempt + 1} failed:`, error);
                        attempt++;
                        if (attempt < maxAttempts) {
                            // Implementa retroceso exponencial (1s, 2s, 4s)
                            await new Promise(resolve => setTimeout(resolve, Math.pow(2, attempt) * 1000));
                        }
                    }
                }

                setRewardMessage(responseText);
                setIsLoading(false);
            };

            const closeModal = () => {
                setIsModalOpen(false);
                setRewardMessage('');
            };

            return (
                <div
                    className={`relative w-screen h-screen flex flex-col items-center justify-center
                    transition-all duration-500 ease-in-out font-inter ${backgroundColor} text-white`}
                >
                    {/* Botón de Configuración */}
                    <button
                        className="absolute top-4 right-4 p-3 bg-white/20 hover:bg-white/30 transition-colors
                        rounded-full shadow-lg z-20 focus:outline-none focus:ring-4 focus:ring-white/50"
                        onClick={() => setIsSettingsOpen(prev => !prev)}
                        aria-label="Abrir Configuraciones"
                    >
                        <SettingsIcon className="w-6 h-6 text-white" />
                    </button>

                    {/* Menú de Configuración Desplegable */}
                    {isSettingsOpen && (
                        <div className="absolute top-16 right-4 w-64 bg-gray-900/95 backdrop-blur-sm p-4
                            rounded-xl shadow-2xl z-10 space-y-4 border border-gray-700/50">
                            <h3 className="text-xl font-bold mb-2 border-b border-gray-700 pb-2">Configuraciones</h3>

                            {/* Opción 1: Obtener Recompensa/Meta (Gemini Feature) */}
                            <button
                                onClick={() => getAIPrompt(count)}
                                disabled={isLoading}
                                className="w-full text-left p-3 rounded-lg bg-yellow-500 hover:bg-yellow-400
                                transition-all shadow-md flex items-center justify-center font-bold text-gray-900
                                disabled:opacity-50 disabled:cursor-not-allowed gemini-shadow"
                            >
                                <StarIcon className="w-5 h-5 mr-2" />
                                {isLoading ? 'Generando...' : 'Meta de la IA ✨'}
                            </button>

                            {/* Opción 2: Reiniciar Contador */}
                            <button
                                onClick={handleReset}
                                className="w-full text-left p-3 rounded-lg bg-indigo-600 hover:bg-indigo-700
                                transition-all shadow-md flex items-center justify-center font-medium"
                            >
                                <svg xmlns="http://www.w3.org/2000/svg" className="w-5 h-5 mr-2" viewBox="0 0 24 24"
                                    fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round"
                                    strokeLinejoin="round"><path d="M3 12a9 9 0 0 1 9-9 9.75 9.75 0 0 1 6.74 2.76L21 6"/><path
                                    d="M21 3v3h-3"/><path d="M21 12a9 9 0 0 1-9 9 9.75 9.75 0 0 1-6.74-2.76L3 18"/><path
                                    d="M3 21v-3h3"/></svg>
                                Reiniciar Contador
                            </button>

                            {/* Opción 3: Alternar Cambio de Color Automático */}
                            <div className="flex items-center justify-between p-3 bg-gray-800 rounded-lg shadow-inner">
                                <span className="font-medium text-sm">Cambio de color al hacer clic</span>
                                <button
                                    onClick={handleToggleColorChange}
                                    className={`relative inline-flex h-6 w-11 items-center rounded-full transition-colors
                                    focus:outline-none focus:ring-2 focus:ring-offset-2 ${
                                        isColorChangeEnabled ? 'bg-green-500 focus:ring-green-500' : 'bg-red-500 focus:ring-red-500'
                                    }`}
                                    aria-checked={isColorChangeEnabled}
                                    role="switch"
                                >
                                    <span
                                        className={`inline-block h-4 w-4 transform rounded-full bg-white transition-transform ${
                                            isColorChangeEnabled ? 'translate-x-6' : 'translate-x-1'
                                        }`}
                                    />
                                </button>
                            </div>
                            <p className="text-xs text-gray-400 pt-2">Fondo dinámico: <span
                            className="font-bold">{isColorChangeEnabled ? 'Activado' : 'Inactivo'}</span></p>
                        </div>
                    )}

                    {/* Área del Contador Principal */}
                    <div
                        className="text-center cursor-pointer p-8 rounded-2xl shadow-2xl backdrop-blur-sm
                        bg-white/10 hover:bg-white/20 transition-all duration-300 transform active:scale-95"
                        onClick={handleIncrement}
                        role="button"
                        tabIndex="0"
                        aria-label={`El contador actual es ${count}. Haz clic para incrementar.`}
                    >
                        <p className="text-2xl font-light opacity-80 mb-2">Haz clic para sumar:</p>
                        <h1 className="text-9xl font-extrabold tracking-tight select-none">
                            {count}
                        </h1>
                    </div>
                    
                    {/* Mensaje de estado de color */}
                    <p className="absolute bottom-4 text-sm font-medium opacity-70">
                        Fondo dinámico: {isColorChangeEnabled ? 'Activado' : 'Desactivado'}
                    </p>

                    {/* Modal para el Mensaje de la IA */}
                    {isModalOpen && (
                        <div className="fixed inset-0 bg-black/70 z-30 flex items-center justify-center p-4" onClick={closeModal}>
                            <div className="bg-gray-800 text-white p-6 rounded-xl shadow-2xl w-full max-w-md
                                border-2 border-yellow-500 animate-in fade-in zoom-in-50"
                                onClick={(e) => e.stopPropagation()}>
                                <h3 className="text-2xl font-extrabold text-yellow-400 mb-3 flex items-center">
                                    <StarIcon className="w-6 h-6 mr-2" fill="#FBBF24" />
                                    {isLoading ? 'Conectando...' : '¡Meta de la IA!'}
                                </h3>
                                <div className="text-sm text-gray-300 min-h-[50px]">
                                    {isLoading ? (
                                        <div className="flex items-center space-x-2">
                                            <div className="animate-spin rounded-full h-4 w-4 border-b-2 border-yellow-400"></div>
                                            <span>Generando un mensaje lúdico...</span>
                                        </div>
                                    ) : (
                                        <p style={{ whiteSpace: 'pre-wrap' }}>{rewardMessage}</p>
                                    )}
                                </div>
                                <button
                                    onClick={closeModal}
                                    className="mt-4 w-full p-2 bg-indigo-600 hover:bg-indigo-700 rounded-lg font-medium transition-colors"
                                >
                                    Cerrar
                                </button>
                            </div>
                        </div>
                    )}

                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);

    </script>

</body>
</html>
